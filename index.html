const video = document.getElementById("video"); const button = document.getElementById("snap"); const debug = document.getElementById("debug"); let slimes = []; let baseScale = 3.0; /* 🟢 创建3个史莱姆 */ function createSlimes() { for (let i = 0; i < 3; i++) { let el = document.createElement("div"); el.innerHTML = document.getElementById("slime").outerHTML; document.body.appendChild(el.firstElementChild); slimes.push({ el: document.querySelectorAll("#slime")[i], x: window.innerWidth * (0.3 + i * 0.2), y: window.innerHeight * 0.5, vx: 0, vy: 0 }); } } /* 🟢 touch控制（控制第一个） */ window.addEventListener("touchmove", (e) => { let t = e.touches[0]; let dx = t.clientX - window.innerWidth / 2; let dy = t.clientY - window.innerHeight / 2; slimes[0].vx += dx * 0.0008; slimes[0].vy += dy * 0.0008; }); /* 🟢 碰撞 + 物理 */ function update() { for (let i = 0; i < slimes.length; i++) { let a = slimes[i]; a.vx *= 0.92; a.vy *= 0.92; a.x += a.vx; a.y += a.vy; } /* 🟢 碰撞检测 */ for (let i = 0; i < slimes.length; i++) { for (let j = i + 1; j < slimes.length; j++) { let a = slimes[i]; let b = slimes[j]; let dx = b.x - a.x; let dy = b.y - a.y; let dist = Math.sqrt(dx*dx + dy*dy); let minDist = 160; if (dist < minDist) { let force = (minDist - dist) * 0.02; let nx = dx / dist; let ny = dy / dist; a.vx -= nx * force; a.vy -= ny * force; b.vx += nx * force; b.vy += ny * force; /* 🟢 形变（挤压感） */ a.el.style.transform += ` scale(1.2,0.8)`; b.el.style.transform += ` scale(1.2,0.8)`; } } } /* 🟢 渲染 */ for (let i = 0; i < slimes.length; i++) { let s = slimes[i]; s.x = Math.max(80, Math.min(window.innerWidth - 80, s.x)); s.y = Math.max(80, Math.min(window.innerHeight - 80, s.y)); let angle = Math.atan2(s.vy, s.vx); let stretchX = 1 + s.vx * 0.02; let stretchY = 1 - s.vy * 0.02; s.el.style.transform = `translate(${s.x - 110}px, ${s.y - 110}px) rotate(${angle}rad) scale(${baseScale * stretchX}, ${baseScale * stretchY})`; } requestAnimationFrame(update); } /* 🟢 摄像头 */ async function startCamera() { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" }, audio: false }); video.srcObject = stream; } /* 🟢 启动 */ button.onclick = async () => { await startCamera(); createSlimes(); update(); };
